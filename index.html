-- Execute no SQL Editor do Supabase: seu projeto → SQL Editor → New query -- 1. Criar tabela CREATE TABLE credenciamentos ( id BIGSERIAL PRIMARY KEY, municipio TEXT NOT NULL, nome TEXT NOT NULL, documento TEXT NOT NULL, cargo TEXT NOT NULL, assinatura TEXT, consentimento BOOLEAN DEFAULT TRUE, created_at TIMESTAMPTZ DEFAULT NOW() ); -- 2. Segurança por linha (RLS) ALTER TABLE credenciamentos ENABLE ROW LEVEL SECURITY; -- 3. Permitir leitura pública (exibir quem já credenciou) CREATE POLICY "Leitura publica" ON credenciamentos FOR SELECT USING (true); -- 4. Permitir inserção pública (registrar credenciamento) CREATE POLICY "Insercao publica" ON credenciamentos FOR INSERT WITH CHECK (true); -- 5. Ativar atualização em tempo real ALTER PUBLICATION supabase_realtime ADD TABLE credenciamentos;
